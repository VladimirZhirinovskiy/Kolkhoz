add_namespace = political

# Communism on the Rise
country_event = {
	id = political.1
	title = political.1.t
	desc = political.1.d
	picture = GFX_report_event_worker_protests

 	trigger = {
		NOT = { has_government = communist }
		communist > 0.5
	}

	fire_only_once = yes
	
	mean_time_to_happen = {
		days = 30
	}




	option = {
		name = political.1.a
		add_political_power = -100
		set_politics = {
			ruling_party = communist
			elections_allowed = no
		}
	}
	option = {
		name = political.1.b
		add_stability = -0.15
		add_political_power = -200
		add_popularity = {
			ideology = communist
			popularity = -0.10
		}
		
		
	}
	
}
